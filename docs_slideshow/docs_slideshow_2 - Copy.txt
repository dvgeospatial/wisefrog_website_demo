    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const slides = document.querySelectorAll('.slide');
            const captions = document.querySelectorAll('.caption');
            let currentIndex = 0;
            const displayTime = 3000; // 3 seconds per slide
            
            setInterval(() => {
                // Identify current items
                const currentSlide = slides[currentIndex];
                const currentCaption = captions[currentIndex];

                // Calculate next index
                currentIndex = (currentIndex + 1) % slides.length;
                
                const nextSlide = slides[currentIndex];
                const nextCaption = captions[currentIndex];

                // Remove active class (Fade Out)
                currentSlide.classList.remove('active');
                currentCaption.classList.remove('active');
                
                // Add active class (Fade In)
                nextSlide.classList.add('active');
                nextCaption.classList.add('active');

            }, displayTime);
        });
    </script>
